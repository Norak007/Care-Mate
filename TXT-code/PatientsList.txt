    private DefaultListModel<String> namesList;
    private java.util.List<Integer> idsList;
    
    private void loadPatientList() {
        try {
            String sql = "SELECT patient_id,Full_Name FROM Patient"; 
            PreparedStatement pt = con.prepareStatement(sql);
            ResultSet rs = pt.executeQuery();
        
            namesList = new DefaultListModel<>();
            idsList = new java.util.ArrayList<>();
            
            while (rs.next()) {
            namesList.addElement(rs.getString("Full_Name"));
            idsList.add(rs.getInt("patient_id")); // نحفظ الـ ID بنفس الترتيب
        }
        jList1.setModel(namesList);

    } catch (Exception ex) {
        JOptionPane.showMessageDialog(this, "Error: " + ex.getMessage());
    }
}

try {
    Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/caremate2", "root", "2004");

    String sql = "SELECT note_content FROM CaregiverNote WHERE patient_id = ?";
    PreparedStatement pt = con.prepareStatement(sql);
    pt.setInt(1, PatientsInfo.selectedPatientId);

    ResultSet rs = pt.executeQuery();
    model.setRowCount(0); // حذف أي بيانات قديمة

    while (rs.next()) {
        model.addRow(new Object[]{rs.getString("note_content")});
    }

    con.close();
} catch (Exception ex) {

}

    private void loadUpcomingAppointment() {
    try {
        String sql = "SELECT appointment_date, appointment_time, reason FROM Appointment WHERE patient_id = ? ORDER BY appointment_date, appointment_time"; 
        PreparedStatement pt = con.prepareStatement(sql);
        pt.setInt(1, patientId);

        ResultSet rs = pt.executeQuery();
        DefaultListModel<String> listModel = new DefaultListModel<>();

        while (rs.next()) {
            String date = rs.getDate("appointment_date").toString();
            String time = rs.getTime("appointment_time").toString();
            String reason = rs.getString("reason");

            listModel.addElement(date + " at " + time + " - " + reason);
        }

        jList1.setModel(listModel);

    } catch (Exception ex) {
        JOptionPane.showMessageDialog(this, "Error: " + ex.getMessage());
    }
}

    private void loadMedicationDetails() {
        String sql = "SELECT * FROM Medication WHERE medication_name = ?";
        try {
            PreparedStatement pt = con.prepareStatement(sql);
            pt.setString(1, medName);
            ResultSet rs = pt.executeQuery();

            DefaultTableModel model = (DefaultTableModel) jTable1.getModel();
            model.setRowCount(0);

            while (rs.next()) {
                model.addRow(new Object[]{
                    rs.getString("medication_id"),
                    rs.getString("patient_id"),
                    rs.getString("medication_name"),
                    rs.getString("dosage"),
                    rs.getString("medication_time"),
                    rs.getString("start_date"),
                    rs.getString("end_date"),
                    rs.getString("type"),
                    rs.getString("instructions"),
                    rs.getString("last_taken"),
                    rs.getString("next_dose")
                });
            }

        } catch (Exception ex) {
            JOptionPane.showMessageDialog(this, "Error: " + ex.getMessage());
        }
    
    }
